In this case we made use of many values for the background color as they pair for the order in which they are represented in the data array as well as the labels. Just for demonstration pourposes we can also see a function that add values to certain variables belonging to the data. This is a great example of how we can add values dinamically to our chart. 


//REMEMBER: must follow the naming conventios, for this is very important for Chart.js
const ctx1 = document.getElementById("pieChartsTutorial").getContext("2d");

let angularVal = randomVal();
let reactVal = randomVal();
let jqueryVal = randomVal();
let alpineVal = randomVal();
let gPolymerVal = randomVal();
let vueVal = randomVal();

const pieChartsTutorial = new Chart(ctx1, {
    type: "pie", //type of chart: Ï€
    data: {
    labels: ["Angular", "React", "JQuery", "Alpine", "GooglePolymer", "Vue"],
    datasets: [
        {
        //random values here
        data: [angularVal, reactVal, jqueryVal, alpineVal, gPolymerVal, vueVal],
        backgroundColor: [
            "rgba(255, 99, 132, 0.2)", //RGB color code with opacity for the background fill
            "rgba(54, 162, 235, 0.2)", //(thought it looks neat ðŸ¤“ )
            "rgba(255, 206, 86, 0.2)",
            "rgba(75, 192, 192, 0.2)",
            "rgba(153, 102, 255, 0.2)",
            "rgba(255, 159, 64, 0.2)",
        ],
        borderColor: [
            "rgba(255, 99, 132, 1)",
            "rgba(54, 162, 235, 1)",
            "rgba(255, 206, 86, 1)",
            "rgba(75, 192, 192, 1)",
            "rgba(153, 102, 255, 1)",
            "rgba(255, 159, 64, 1)",
        ],
        borderWidth: 1,
        },
    ],
    },
});
//returns a random number between 1 and 19
function randomVal(){
    return  Math.floor(Math.random() * 19);
}
